175 template <class... Args>
176 LogEntry& SolverLogger::AddEntry(const int& col, Args... args) {
177  ALTRO_ASSERT(
178  col <= static_cast<int>(entries_.size()),
179  fmt::format("Column ({}) must be less than or equal to the current number of entries ({}).",
180  col, NumEntries()));
181  ALTRO_ASSERT(
182  col >= -static_cast<int>(entries_.size()) - 1,
183  fmt::format(
184  "Column ({}) must be greater or equal to than the negative new number of entries ({}).",
185  col, -NumEntries() - 1));
186 
187  // Create a LogEntry and forward it to the map after extracting the title
188  LogEntry entry(std::forward<Args>(args)...);
189  const std::string title = entry.GetTitle();
190  auto insert = entries_.emplace(std::make_pair(title, std::move(entry)));
191 
192  // Get a pointer to the title string used in the map
193  // insert.first is an iterator over key-value pairs for the map entries_
194  const std::string* title_ptr = &(insert.first->first);
195 
196  // Specify the output order
197  auto it = order_.begin();
198  if (col < 0) {
199  // Count from the back if the input is negative
200  it = order_.end() + col + 1;
201  } else {
202  it += col;
203  }
204  order_.insert(it, title_ptr);
205 
206  return insert.first->second;
207 }
