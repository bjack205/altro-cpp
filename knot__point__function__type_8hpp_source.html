149  void
150  CalcActionValueExpansion(const Eigen::Ref<const MatrixXd> &ctg_hessian,
151  const Eigen::Ref<const MatrixXd> &ctg_gradient) {
152  Eigen::Block<JacType, n, n> A = dynamics_expansion_.GetA();
153  Eigen::Block<JacType, n, m> B = dynamics_expansion_.GetB();
154  action_value_expansion_.dxdx() =
155  cost_expansion_.dxdx() + A.transpose() * ctg_hessian * A;
156  action_value_expansion_.dxdu() =
157  cost_expansion_.dxdu() + A.transpose() * ctg_hessian * B;
158  action_value_expansion_.dudu() =
159  cost_expansion_.dudu() + B.transpose() * ctg_hessian * B;
160  action_value_expansion_.dx() =
161  cost_expansion_.dx() + A.transpose() * ctg_gradient;
162  action_value_expansion_.du() =
163  cost_expansion_.du() + B.transpose() * ctg_gradient;
164  }

197  Eigen::ComputationInfo CalcGains() {
198  // TODO(bjackson): Store factorization in the class
199  Eigen::LLT<Eigen::Matrix<double, m, m>> Quu_chol;
200  Quu_chol.compute(action_value_expansion_regularized_.dudu());
201  Eigen::ComputationInfo info = Quu_chol.info();
202  if (info == Eigen::Success) {
203  feedback_gain_ = Quu_chol.solve(
204  action_value_expansion_regularized_.dxdu().transpose());
205  feedback_gain_ *= -1;
206  feedforward_gain_ =
207  Quu_chol.solve(action_value_expansion_regularized_.du());
208  feedforward_gain_ *= -1;
209  }
210  return info;
211  }

220  void CalcCostToGo() {
221  Eigen::Matrix<double, m, n> &K = GetFeedbackGain();
222  Eigen::Matrix<double, m, 1> &d = GetFeedforwardGain();
223  CostExpansion<n, m> &Q = GetActionValueExpansion();
224  ctg_gradient_ = Q.dx() + K.transpose() * Q.dudu() * d +
225  K.transpose() * Q.du() + Q.dxdu() * d;
226  ctg_hessian_ = Q.dxdx() + K.transpose() * Q.dudu() * K +
227  K.transpose() * Q.dxdu().transpose() + Q.dxdu() * K;
228  ctg_delta_[0] = d.dot(Q.du());
229  ctg_delta_[1] = 0.5 * d.dot(Q.dudu() * d); // NOLINT(readability-magic-numbers)
230  }
