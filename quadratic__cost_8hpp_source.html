29  static QuadraticCost LQRCost(const MatrixXd& Q, const MatrixXd& R, const VectorXd& xref,
30  const VectorXd& uref, bool terminal = false) {
31  int n = Q.rows();
32  int m = R.rows();
33  ALTRO_ASSERT(xref.size() == n, "xref is the wrong size.");
34  MatrixXd H = MatrixXd::Zero(n, m);
35  VectorXd q = -(Q * xref);
36  VectorXd r = -(R * uref);
37  double c = 0.5 * xref.dot(Q * xref) + 0.5 * uref.dot(R * uref);
38  return QuadraticCost(Q, R, H, q, r, c, terminal);
39  }
