98  void CopyFromProblem(const problem::Problem& prob, int k_start, int k_stop) {
99  ALTRO_ASSERT(0 <= k_start && k_start <= N_,
100  fmt::format("Start index must be in the interval [0,{}]", N_));
101  ALTRO_ASSERT(0 <= k_stop && k_stop <= N_ + 1,
102  fmt::format("Start index must be in the interval [0,{}]", N_ + 1));
103  ALTRO_ASSERT(prob.IsFullyDefined(), "Expected problem to be fully defined.");
104  for (int k = k_start; k < k_stop; ++k) {
105  if (n != Eigen::Dynamic) {
106  ALTRO_ASSERT(
107  prob.GetDynamics(k)->StateDimension() == n,
108  fmt::format("Inconsistent state dimension at knot point {}. Expected {}, got {}", k, n,
109  prob.GetDynamics(k)->StateDimension()));
110  }
111  if (m != Eigen::Dynamic) {
112  ALTRO_ASSERT(
113  prob.GetDynamics(k)->ControlDimension() == m,
114  fmt::format("Inconsistent control dimension at knot point {}. Expected {}, got {}", k,
115  m, prob.GetDynamics(k)->ControlDimension()));
116  }
117  std::shared_ptr<problem::DiscreteDynamics> model = prob.GetDynamics(k);
118  std::shared_ptr<problem::CostFunction> costfun = prob.GetCostFunction(k);
119  knotpoints_.emplace_back(std::make_unique<ilqr::KnotPointFunctions<n2, m2>>(model, costfun));
120  }
121  initial_state_ = prob.GetInitialStatePointer();
122  is_initial_state_set = true;
123  }

386  void BackwardPass() {
387  Stopwatch sw = stats_.GetTimer()->Start("backward_pass");
388 
389  // Regularization
390  Eigen::ComputationInfo info;
391 
392  // Terminal Cost-to-go
393  knotpoints_[N_]->CalcTerminalCostToGo();
394  Eigen::Matrix<double, n, n>* Sxx_prev = &(knotpoints_[N_]->GetCostToGoHessian());
395  Eigen::Matrix<double, n, 1>* Sx_prev = &(knotpoints_[N_]->GetCostToGoGradient());
396 
397  int max_reg_count = 0;
398  deltaV_[0] = 0.0;
399  deltaV_[1] = 0.0;
400 
401  bool repeat_backwardpass = true;
402  while (repeat_backwardpass) {
403  for (int k = N_ - 1; k >= 0; --k) {
404  // TODO(bjackson)[SW-16103] Create a test that checks this
405  knotpoints_[k]->CalcActionValueExpansion(*Sxx_prev, *Sx_prev);
406  knotpoints_[k]->RegularizeActionValue(rho_);
407  info = knotpoints_[k]->CalcGains();
408 
409  // Handle solve failure
410  if (info != Eigen::Success) {
411 
412  IncreaseRegularization();
413 
414  // Reset the cost-to-go pointers to the terminal expansion
415  Sxx_prev = &(knotpoints_[N_]->GetCostToGoHessian());
416  Sx_prev = &(knotpoints_[N_]->GetCostToGoGradient());
417 
418  // Check if we're at max regularization
419  if (rho_ >= GetOptions().bp_reg_max) {
420  max_reg_count++;
421  }
422 
423  if (max_reg_count >= GetOptions().bp_reg_fail_threshold) {
424  status_ = SolverStatus::kBackwardPassRegularizationFailed;
425  repeat_backwardpass = false;
426  }
427  break;
428  }
429 
430  // Update Cost-To-Go
431  knotpoints_[k]->CalcCostToGo();
432  knotpoints_[k]->AddCostToGo(&deltaV_);
433 
434  Sxx_prev = &(knotpoints_[k]->GetCostToGoHessian());
435  Sx_prev = &(knotpoints_[k]->GetCostToGoGradient());
436 
437  // Backward pass successful if it calculates the cost to go at
438  // the first knot point.
439  if (k == 0) {
440  repeat_backwardpass = false;
441  }
442  } // end for
443  } // end while
444  stats_.Log("reg", rho_);
445  DecreaseRegularization();
446  }

469  bool RolloutClosedLoop(const double alpha) {
470  Stopwatch sw = stats_.GetTimer()->Start("rollout");
471 
472  Zbar_->State(0) = *initial_state_;
473  for (int k = 0; k < N_; ++k) {
474  MatrixNxMd<m, n>& K = GetKnotPointFunction(k).GetFeedbackGain();
475  VectorNd<m>& d = GetKnotPointFunction(k).GetFeedforwardGain();
476 
477  // TODO(bjackson): Make this a function of the dynamics
478  VectorNd<n> dx = Zbar_->State(k) - Z_->State(k);
479  Zbar_->Control(k) = Z_->Control(k) + K * dx + d * alpha;
480 
481  // Simulate forward with feedback
482  GetKnotPointFunction(k).Dynamics(Zbar_->State(k), Zbar_->Control(k), Zbar_->GetTime(k),
483  Zbar_->GetStep(k), Zbar_->State(k + 1));
484 
485  if (GetOptions().check_forwardpass_bounds) {
486  if (Zbar_->State(k + 1).norm() > GetOptions().state_max) {
487  // TODO(bjackson): Emit warning (need logging mechanism)
488  status_ = SolverStatus::kStateLimit;
489  return false;
490  }
491  if (Zbar_->Control(k).norm() > GetOptions().control_max) {
492  // TODO(bjackson): Emit warning (need logging mechanism)
493  status_ = SolverStatus::kControlLimit;
494  return false;
495  }
496  }
497  }
498  status_ = SolverStatus::kUnsolved;
499  return true;
500  }

513  void ForwardPass() {
514  Stopwatch sw = stats_.GetTimer()->Start("forward_pass");
515  SolverOptions& opts = GetOptions();
516 
517  double J0 = costs_.sum(); // Calculated during UpdateExpansions
518 
519  double alpha = 1.0;
520  double z = -1.0;
521  int iter_fp = 0;
522  bool success = false;
523 
524  double J = J0;
525 
526  for (; iter_fp < opts.line_search_max_iterations; ++iter_fp) {
527  if (RolloutClosedLoop(alpha)) {
528  J = Cost(*Zbar_);
529  double expected = -alpha * (deltaV_[0] + alpha * deltaV_[1]);
530  if (expected > 0.0) {
531  z = (J0 - J) / expected;
532  } else {
533  z = -1.0;
534  }
535 
536  if (opts.line_search_lower_bound <= z && z <= opts.line_search_upper_bound && J < J0) {
537  success = true;
538  // stats_.improvement_ratio.emplace_back(z);
539  stats_.Log("cost", J);
540  stats_.Log("alpha", alpha);
541  stats_.Log("z", z);
542  break;
543  }
544  }
545  alpha /= opts.line_search_decrease_factor;
546  }
547 
548  if (success) {
549  (*Z_) = (*Zbar_);
550  } else {
551  IncreaseRegularization();
552  J = J0;
553  }
554 
555  if (J > J0) {
556  // TODO(bjackson): Emit warning (needs logging)
557  status_ = SolverStatus::kCostIncrease;
558  }
559  }

708  void LaunchThreads() {
709  size_t nthreads = GetOptions().NumThreads();
710 
711  // Reset the thread pool if the requested number of threads changes
712  int threadpool_size = NumThreads();
713  bool single_threaded = threadpool_size == 0 && nthreads == 1;
714  if (single_threaded) {
715  return;
716  }
717  bool num_threads_changed = nthreads != NumThreads();
718  if (N_ > 0 && (num_threads_changed || ShouldRedoTaskAssignment())) {
719  if (pool_.IsRunning()) {
720  pool_.StopThreads();
721  }
722  tasks_.clear();
723 
724  // Create tasks
725  std::vector<int>& work_inds = GetTaskAssignment();
726  int ntasks = NumTasks();
727  for (int i = 0; i < ntasks; ++i) {
728  int start = work_inds[i];
729  int stop = work_inds[i + 1];
730  auto expansion_block = [this, start, stop]() { UpdateExpansionsBlock(start, stop); };
731  tasks_.emplace_back(std::move(expansion_block));
732  }
733 
734  // Start the pool
735  if (nthreads > 1) {
736  pool_.LaunchThreads(nthreads);
