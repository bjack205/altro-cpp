11 Eigen::Matrix<double, nrows, ncols> FiniteDiffJacobian(
12  const Func &f, const Eigen::Ref<const Eigen::Matrix<double, ncols, 1>> &x,
13  const double eps = 1e-6, const bool central = false) {
14  const int n = x.rows();
15 
16  // Evaluate the function and get output size
17  Eigen::Matrix<double, nrows, 1> y = f(x);
18  const int m = y.rows();
19  Eigen::Matrix<double, nrows, ncols> jac =
20  Eigen::Matrix<double, nrows, ncols>::Zero(m, n);
21 
22  // Create pertubation vector
23  Eigen::Matrix<double, ncols, 1> e = Eigen::Matrix<double, ncols, 1>::Zero(n);
24 
25  // Loop over columns
26  e(0) = eps;
27  for (int i = 0; i < n; ++i) {
28  double step = eps;
29  if (central) {
30  y = f(x - e);
31  step = 2 * eps;
32  }
33  jac.col(i) = (f(x + e) - y) / step;
34  if (i < n - 1) {
35  e(i + 1) = e(i);
36  e(i) = 0;
37  }
38  }
39  return jac;
40 }
