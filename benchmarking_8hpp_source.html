46 template <class Duration>
47 BenchmarkResults<Duration> BenchmarkResults<Duration>::Calculate(std::vector<time_t>& times) {
48  // Sort for median
49  std::sort(times.begin(), times.end());
50  int samples = times.size();
51  time_t median;
52  if ((samples % 2) == 1) { // odd
53  median = times[samples / 2];
54  } else {
55  median = (times[samples / 2] + times[(samples / 2) - 1]) / 2;
56  }
57  time_t sum = times[0];
58  time_t max = times[0];
59  time_t min = times[0];
60  for (int i = 1; i < samples; ++i) {
61  sum += times[i];
62  max = std::max(times[i], max);
63  min = std::min(times[i], min);
64  }
65  time_t mean = sum / samples;
66  time_t std = time_t(std::pow((times[0] - mean).count(), 2));
67  for (int i = 1; i < samples; ++i) {
68  std += time_t(std::pow((times[i] - mean).count(), 2));
69  }
70  std = time_t(std::sqrt((std / samples).count()));
71  BenchmarkResults<Duration> res{mean, median, std, max, min, samples};
72  return res;
73 }
