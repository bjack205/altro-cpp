86  std::vector<constraints::ConstraintInfo> GetConstraintInfo(bool should_sort = false) const {
87  std::vector<constraints::ConstraintInfo> coninfo;
88  int i_last = 0;
89  for (int k = 0; k <= NumSegments(); ++k) {
90  costs_[k]->GetConstraintInfo(&coninfo);
91  for (int i = i_last; i < static_cast<int>(coninfo.size()); ++i) {
92  coninfo[i].index = k;
93  }
94  i_last = coninfo.size();
95  }
96  if (should_sort) {
97  auto comp = [](const constraints::ConstraintInfo& info1,
98  const constraints::ConstraintInfo& info2) {
99  return info1.violation.lpNorm<Eigen::Infinity>() > info2.violation.lpNorm<Eigen::Infinity>();
100  };
101  std::sort(coninfo.begin(), coninfo.end(), comp);
102  }
103  return coninfo;
104  }

368 template <int n, int m>
369 bool AugmentedLagrangianiLQR<n, m>::IsDone() {
370  Stopwatch sw = CreateTimer("convergence_check");
371 
372  SolverStats& stats = GetStats();
373  SolverOptions& opts = GetOptions();
374  const bool are_constraints_satisfied = stats.violations.back() < opts.constraint_tolerance;
375  const bool is_max_penalty_exceeded = stats.max_penalty.back() > opts.maximum_penalty;
376  const bool is_max_outer_iterations_exceeded = stats.iterations_outer >= opts.max_iterations_outer;
377  const bool is_max_total_iterations_exeeded = stats.iterations_total >= opts.max_iterations_total;
378  if (ilqr_solver_.GetStatus() != SolverStatus::kSolved) {
379  status_ = ilqr_solver_.GetStatus();
380  return true;
381  }
382  if (are_constraints_satisfied) {
383  if (ilqr_solver_.GetStatus() == SolverStatus::kSolved) {
384  status_ = SolverStatus::kSolved;
385  return true;
386  }
387  }
388  if (is_max_penalty_exceeded) {
389  status_ = SolverStatus::kMaxPenalty;
390  return true;
391  }
392  if (is_max_outer_iterations_exceeded) {
393  status_ = SolverStatus::kMaxOuterIterations;
394  return true;
395  }
396  if (is_max_total_iterations_exeeded) {
397  status_ = SolverStatus::kMaxIterations;
398  return true;
399  }
400  return false;
401 }
